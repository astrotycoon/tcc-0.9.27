140/* copy a string and truncate it. */
141ST_FUNC char *pstrcpy(char *buf, int buf_size, const char *s)
142{
143 char *q, *q_end;
144 int c;
145
146 if (buf_size > 0) {
147 q = buf;
148 q_end = buf + buf_size - 1;
149 while (q < q_end) {
150 c = *s++;
151 if (c == '\0')
152 break;
153 *q++ = c;
154 }
155 *q = '\0';
156 }
157 return buf;
158}

160/* strcat and truncate. */
161ST_FUNC char *pstrcat(char *buf, int buf_size, const char *s)
162{
163 int len;
164 len = strlen(buf);
165 if (len < buf_size)
166 pstrcpy(buf + len, buf_size - len, s);
167 return buf;
168}

255#define MEM_DEBUG_CHECK3(header) \

262struct mem_debug_header {
263 unsigned magic1;
264 unsigned size;
265 struct mem_debug_header *prev;
266 struct mem_debug_header *next;
267 int line_num;
268 char file_name[MEM_DEBUG_FILE_LEN + 1];
269 unsigned magic2;
270 ALIGNED(16) unsigned magic3;

275static mem_debug_header_t *mem_debug_chain;

278
279static mem_debug_header_t *malloc_check(void *ptr, const char *msg)
280{
281 mem_debug_header_t * header = MEM_HEADER_PTR(ptr);
282 if (header->magic1 != MEM_DEBUG_MAGIC1 ||
283 header->magic2 != MEM_DEBUG_MAGIC2 ||
284 MEM_DEBUG_CHECK3(header) != MEM_DEBUG_MAGIC3 ||
285 header->size == (unsigned)-1) {
286 fprintf(stderr, "%s check failed\n", msg);
287 if (header->magic1 == MEM_DEBUG_MAGIC1)
288 fprintf(stderr, "%s:%u: block allocated here.\n",
289 header->file_name, header->line_num);
290 exit(1);
291 }
292 return header;
293}

295PUB_FUNC void *tcc_malloc_debug(unsigned long size, const char *file, int line)
296{
297 int ofs;
298 mem_debug_header_t *header;
299
300 header = malloc(sizeof(mem_debug_header_t) + size);
301 if (!header)
302 tcc_error("memory full (malloc)");
303
304 header->magic1 = MEM_DEBUG_MAGIC1;
305 header->magic2 = MEM_DEBUG_MAGIC2;
306 header->size = size;
307 MEM_DEBUG_CHECK3(header) = MEM_DEBUG_MAGIC3;
308 header->line_num = line;
309 ofs = strlen(file) - MEM_DEBUG_FILE_LEN;
310 strncpy(header->file_name, file + (ofs > 0 ? ofs : 0), MEM_DEBUG_FILE_LEN);
311 header->file_name[MEM_DEBUG_FILE_LEN] = 0;
312
313 header->next = mem_debug_chain;
314 header->prev = NULL;
315 if (header->next)
316 header->next->prev = header;
317 mem_debug_chain = header;
318
319 mem_cur_size += size;
320 if (mem_cur_size > mem_max_size)
321 mem_max_size = mem_cur_size;
322
323 return MEM_USER_PTR(header);
324}

326PUB_FUNC void tcc_free_debug(void *ptr)
327{
328 mem_debug_header_t *header;
329 if (!ptr)
330 return;
331 header = malloc_check(ptr, "tcc_free");
332 mem_cur_size -= header->size;
333 header->size = (unsigned)-1;
334 if (header->next)
335 header->next->prev = header->prev;
336 if (header->prev)
337 header->prev->next = header->next;
338 if (header == mem_debug_chain)
339 mem_debug_chain = header->next;
340 free(header);
341}

351PUB_FUNC void *tcc_realloc_debug(void *ptr, unsigned long size, const char *file, int line)
352{
353 mem_debug_header_t *header;
354 int mem_debug_chain_update = 0;
355 if (!ptr)
356 return tcc_malloc_debug(size, file, line);
357 header = malloc_check(ptr, "tcc_realloc");
358 mem_cur_size -= header->size;
359 mem_debug_chain_update = (header == mem_debug_chain);
360 header = realloc(header, sizeof(mem_debug_header_t) + size);
361 if (!header)
362 tcc_error("memory full (realloc)");
363 header->size = size;
364 MEM_DEBUG_CHECK3(header) = MEM_DEBUG_MAGIC3;
365 if (header->next)
366 header->next->prev = header;
367 if (header->prev)
368 header->prev->next = header;
369 if (mem_debug_chain_update)
370 mem_debug_chain = header;
371 mem_cur_size += size;
372 if (mem_cur_size > mem_max_size)
373 mem_max_size = mem_cur_size;
374 return MEM_USER_PTR(header);
375}

410ST_FUNC void dynarray_add(void *ptab, int *nb_ptr, void *data)
411{
412 int nb, nb_alloc;
413 void **pp;
414
415 nb = *nb_ptr;
416 pp = *(void ***)ptab;
417 /* every power of two we double array size */
418 if ((nb & (nb - 1)) == 0) {
419 if (!nb)
420 nb_alloc = 1;
421 else
422 nb_alloc = nb * 2;
423 pp = tcc_realloc(pp, nb_alloc * sizeof(void *));
424 *(void***)ptab = pp;
425 }
426 pp[nb++] = data;
427 *nb_ptr = nb;
428}
429
430ST_FUNC void dynarray_reset(void *pp, int *n)
431{
432 void **p;
433 for (p = *(void***)pp; *n; ++p, --*n)
434 if (*p)
435 tcc_free(*p);
436 tcc_free(*(void**)pp);
437 *(void***)pp = NULL;
438}

440static void tcc_split_path(TCCState *s, void *p_ary, int *p_nb_ary, const char *in)
441{
442 const char *p;
443 do {
444 int c;
445 CString str;
446
447 cstr_new(&str);
448 for (p = in; c = *p, c != '\0' && c != PATHSEP[0]; ++p) {
449 if (c == '{' && p[1] && p[2] == '}') {
450 c = p[1], p += 2;
451 if (c == 'B')
452 cstr_cat(&str, s->tcc_lib_path, -1);
453 } else {
454 cstr_ccat(&str, c);
455 }
456 }
457 if (str.size) {
458 cstr_ccat(&str, '\0');
459 dynarray_add(p_ary, p_nb_ary, tcc_strdup(str.data));
460 }
461 cstr_free(&str);
462 in = p+1;
463 } while (*p);
464}

483static void error1(TCCState *s1, int is_warning, const char *fmt, va_list ap)
484{
485 char buf[2048];
486 BufferedFile **pf, *f;
487
488 buf[0] = '\0';
489 /* use upper file if inline ":asm:" or token ":paste:" */
490 for (f = file; f && f->filename[0] == ':'; f = f->prev)
491 ;
492 if (f) {
493 for(pf = s1->include_stack; pf < s1->include_stack_ptr; pf++)
494 strcat_printf(buf, sizeof(buf), "In file included from %s:%d:\n",
495 (*pf)->filename, (*pf)->line_num);
496 if (s1->error_set_jmp_enabled) {
497 strcat_printf(buf, sizeof(buf), "%s:%d: ",
498 f->filename, f->line_num - !!(tok_flags & TOK_FLAG_BOL));
499 } else {
500 strcat_printf(buf, sizeof(buf), "%s: ",
501 f->filename);
502 }
503 } else {
504 strcat_printf(buf, sizeof(buf), "tcc: ");
505 }
506 if (is_warning)
507 strcat_printf(buf, sizeof(buf), "warning: ");
508 else
509 strcat_printf(buf, sizeof(buf), "error: ");
510 strcat_vprintf(buf, sizeof(buf), fmt, ap);
511
512 if (!s1->error_func) {
513 /* default case: stderr */
514 if (s1->output_type == TCC_OUTPUT_PREPROCESS && s1->ppfp == stdout)
515 /* print a newline during tcc -E */
516 printf("\n"), fflush(stdout);
517 fflush(stdout); /* flush -v output */
518 fprintf(stderr, "%s\n", buf);
519 fflush(stderr); /* print error/warning now (win32) */
520 } else {
521 s1->error_func(s1->error_opaque, buf);
522 }
523 if (!is_warning || s1->warn_error)
524 s1->nb_errors++;
525}

578ST_FUNC void tcc_open_bf(TCCState *s1, const char *filename, int initlen)
579{
580 BufferedFile *bf;
581 int buflen = initlen ? initlen : IO_BUF_SIZE;
582
583 bf = tcc_mallocz(sizeof(BufferedFile) + buflen);
584 bf->buf_ptr = bf->buffer;
585 bf->buf_end = bf->buffer + initlen;
586 bf->buf_end[0] = CH_EOB; /* put eob symbol */
587 pstrcpy(bf->filename, sizeof(bf->filename), filename);
588 bf->true_filename = bf->filename;
589 bf->line_num = 1;
590 bf->ifdef_stack_ptr = s1->ifdef_stack_ptr;
591 bf->fd = -1;
592 bf->prev = file;
593 file = bf;
594 tok_flags = TOK_FLAG_BOL | TOK_FLAG_BOF;
595}

597ST_FUNC void tcc_close(void)
598{
599 BufferedFile *bf = file;
600 if (bf->fd > 0) {
601 close(bf->fd);
602 total_lines += bf->line_num;
603 }
604 if (bf->true_filename != bf->filename)
605 tcc_free(bf->true_filename);
606 file = bf->prev;
607 tcc_free(bf);
608}

610ST_FUNC int tcc_open(TCCState *s1, const char *filename)
611{
612 int fd;
613 if (strcmp(filename, "-") == 0)
614 fd = 0, filename = "<stdin>";
615 else
616 fd = open(filename, O_RDONLY | O_BINARY);
617 if ((s1->verbose == 2 && fd >= 0) || s1->verbose == 3)
618 printf("%s %*s%s\n", fd < 0 ? "nf":"->",
619 (int)(s1->include_stack_ptr - s1->include_stack), "", filename);
620 if (fd < 0)
621 return -1;
622 tcc_open_bf(s1, filename, 0);
623#ifdef _WIN32
624 normalize_slashes(file->filename);
625#endif
626 file->fd = fd;
627 return fd;
628}

630/* compile the file opened in 'file'. Return non zero if errors. */
631static int tcc_compile(TCCState *s1)
632{
633 Sym *define_start;
634 int filetype, is_asm;
635
636 define_start = define_stack;
637 filetype = s1->filetype;
638 is_asm = filetype == AFF_TYPE_ASM || filetype == AFF_TYPE_ASMPP;
639 tccelf_begin_file(s1);
640
641 if (setjmp(s1->error_jmp_buf) == 0) {
642 s1->nb_errors = 0;
643 s1->error_set_jmp_enabled = 1;
644
645 preprocess_start(s1, is_asm);
646 if (s1->output_type == TCC_OUTPUT_PREPROCESS) {
647 tcc_preprocess(s1);
648 } else if (is_asm) {
649#ifdef CONFIG_TCC_ASM
650 tcc_assemble(s1, filetype == AFF_TYPE_ASMPP);
651#else
652 tcc_error_noabort("asm not supported");
653#endif
654 } else {
655 tccgen_compile(s1);
656 }
657 }
658 s1->error_set_jmp_enabled = 0;
659
660 preprocess_end(s1);
661 free_inline_functions(s1);
662 /* reset define stack, but keep -D and built-ins */
663 free_defines(define_start);
664 sym_pop(&global_stack, NULL, 0);
665 sym_pop(&local_stack, NULL, 0);
666 tccelf_end_file(s1);
667 return s1->nb_errors != 0 ? -1 : 0;
668}

682/* define a preprocessor symbol. A value can also be provided with the '=' operator */
683LIBTCCAPI void tcc_define_symbol(TCCState *s1, const char *sym, const char *value)
684{
685 int len1, len2;
686 /* default value */
687 if (!value)
688 value = "1";
689 len1 = strlen(sym);
690 len2 = strlen(value);
691
692 /* init file structure */
693 tcc_open_bf(s1, "<define>", len1 + len2 + 1);
694 memcpy(file->buffer, sym, len1);
695 file->buffer[len1] = ' ';
696 memcpy(file->buffer + len1 + 1, value, len2);
697
698 /* parse with define parser */
699 next_nomacro();
700 parse_define();
701 tcc_close();
702}

716/* cleanup all static data used during compilation */
717static void tcc_cleanup(void)
718{
719 if (NULL == tcc_state)
720 return;
721 while (file)
722 tcc_close();
723 tccpp_delete(tcc_state);
724 tcc_state = NULL;
725 /* free sym_pools */

731LIBTCCAPI TCCState *tcc_new(void)
732{
733 TCCState *s;
734
735 tcc_cleanup();
736
737 s = tcc_mallocz(sizeof(TCCState));
738 if (!s)
739 return NULL;
740 tcc_state = s;
741 ++nb_states;
742
743 s->alacarte_link = 1;
744 s->nocommon = 1;
745 s->warn_implicit_function_declaration = 1;
746 s->ms_extensions = 1;
747
748#ifdef CHAR_IS_UNSIGNED
749 s->char_is_unsigned = 1;
750#endif
751#ifdef TCC_TARGET_I386
752 s->seg_size = 32;
753#endif
754 /* enable this if you want symbols with leading underscore on windows: */
755#if 0 /* def TCC_TARGET_PE */
756 s->leading_underscore = 1;
757#endif
758#ifdef _WIN32
759 tcc_set_lib_path_w32(s);
760#else
761 tcc_set_lib_path(s, CONFIG_TCCDIR);
762#endif
763 tccelf_new(s);
764 tccpp_new(s);

1004ST_FUNC int tcc_add_file_internal(TCCState *s1, const char *filename, int flags)
1005{
1006 int ret;
1007
1008 /* open the file */
1009 ret = tcc_open(s1, filename);
1010 if (ret < 0) {
1011 if (flags & AFF_PRINT_ERROR)
1012 tcc_error_noabort("file '%s' not found", filename);
1013 return ret;
1014 }
1015
1016 /* update target deps */
1017 dynarray_add(&s1->target_deps, &s1->nb_target_deps,
1018 tcc_strdup(filename));
1019
1020 if (flags & AFF_TYPE_BIN) {
1021 ElfW(Ehdr) ehdr;
1022 int fd, obj_type;
1023
1024 fd = file->fd;
1025 obj_type = tcc_object_type(fd, &ehdr);
1026 lseek(fd, 0, SEEK_SET);
1027
1028#ifdef TCC_TARGET_MACHO
1029 if (0 == obj_type && 0 == strcmp(tcc_fileextension(filename), ".dylib"))
1030 obj_type = AFF_BINTYPE_DYN;
1031#endif
1032
1033 switch (obj_type) {
1034 case AFF_BINTYPE_REL:
1035 ret = tcc_load_object_file(s1, fd, 0);
1036 break;
1037#ifndef TCC_TARGET_PE
1038 case AFF_BINTYPE_DYN:
1039 if (s1->output_type == TCC_OUTPUT_MEMORY) {
1040 ret = 0;
1041#ifdef TCC_IS_NATIVE
1042 if (NULL == dlopen(filename, RTLD_GLOBAL | RTLD_LAZY))
1043 ret = -1;
1044#endif
1045 } else {
1046 ret = tcc_load_dll(s1, fd, filename,
1047 (flags & AFF_REFERENCED_DLL) != 0);
1048 }
1049 break;
1050#endif
1051 case AFF_BINTYPE_AR:
1052 ret = tcc_load_archive(s1, fd);
1053 break;
1054#ifdef TCC_TARGET_COFF
1055 case AFF_BINTYPE_C67:
1056 ret = tcc_load_coff(s1, fd);
1057 break;
1058#endif
1059 default:
1060#ifdef TCC_TARGET_PE
1061 ret = pe_load_file(s1, filename, fd);
1062#else
1063 /* as GNU ld, consider it is an ld script if not recognized */
1064 ret = tcc_load_ldscript(s1);
1065#endif
1066 if (ret < 0)
1067 tcc_error_noabort("unrecognized file type");
1068 break;
1069 }
1070 } else {
1071 ret = tcc_compile(s1);
1072 }
1073 tcc_close();
1074 return ret;
1075}

1094 } else {
1095 filetype = AFF_TYPE_C;
1096 }
1097 s->filetype = filetype;
1098 }

1245static int strstart(const char *val, const char **str)
1246{
1247 const char *p, *q;
1248 p = *str;
1249 q = val;
1250 while (*q) {
1251 if (*p != *q)
1252 return 0;
1253 p++;
1254 q++;
1255 }
1256 *str = p;
1257 return 1;

1260/* Like strstart, but automatically takes into account that ld options can
1261 *
1262 * - start with double or single dash (e.g. '--soname' or '-soname')
1263 * - arguments can be given as separate or after '=' (e.g. '-Wl,-soname,x.so'
1264 * or '-Wl,-soname=x.so')
1265 *
1266 * you provide `val` always in 'option[=]' form (no leading -)
1267 */
1268static int link_option(const char *str, const char *val, const char **ptr)
1269{
1270 const char *p, *q;
1271 int ret;
1272
1273 /* there should be 1 or 2 dashes */
1274 if (*str++ != '-')
1275 return 0;
1276 if (*str == '-')
1277 str++;
1278
1279 /* then str & val should match (potentially up to '=') */
1280 p = str;
1281 q = val;
1282
1283 ret = 1;
1284 if (q[0] == '?') {
1285 ++q;
1286 if (no_flag(&p))
1287 ret = -1;
1288 }
1289
1290 while (*q != '\0' && *q != '=') {
1291 if (*p != *q)
1292 return 0;
1293 p++;
1294 q++;
1295 }
1296
1297 /* '=' near eos means ',' or '=' is ok */
1298 if (*q == '=') {
1299 if (*p == 0)
1300 *ptr = p;
1301 if (*p != ',' && *p != '=')
1302 return 0;
1303 p++;
1304 } else if (*p) {
1305 return 0;
1306 }
1307 *ptr = p;
1308 return ret;
1309}

1330/* set linker options */
1331static int tcc_set_linker(TCCState *s, const char *option)
1332{
1333 while (*option) {
1334
1335 const char *p = NULL;
1336 char *end = NULL;
1337 int ignoring = 0;
1338 int ret;
1339
1340 if (link_option(option, "Bsymbolic", &p)) {
1341 s->symbolic = 1;
1342 } else if (link_option(option, "nostdlib", &p)) {
1343 s->nostdlib = 1;
1344 } else if (link_option(option, "fini=", &p)) {
1345 copy_linker_arg(&s->fini_symbol, p, 0);
1346 ignoring = 1;
1347 } else if (link_option(option, "image-base=", &p)
1348 || link_option(option, "Ttext=", &p)) {
1349 s->text_addr = strtoull(p, &end, 16);
1350 s->has_text_addr = 1;
1351 } else if (link_option(option, "init=", &p)) {
1352 copy_linker_arg(&s->init_symbol, p, 0);
1353 ignoring = 1;
1354 } else if (link_option(option, "oformat=", &p)) {
1355#if defined(TCC_TARGET_PE)
1356 if (strstart("pe-", &p)) {
1357#elif PTR_SIZE == 8
1358 if (strstart("elf64-", &p)) {
1359#else
1360 if (strstart("elf32-", &p)) {
1361#endif
1362 s->output_format = TCC_OUTPUT_FORMAT_ELF;
1363 } else if (!strcmp(p, "binary")) {
1364 s->output_format = TCC_OUTPUT_FORMAT_BINARY;
1365#ifdef TCC_TARGET_COFF
1366 } else if (!strcmp(p, "coff")) {
1367 s->output_format = TCC_OUTPUT_FORMAT_COFF;
1368#endif
1369 } else
1370 goto err;
1371
1372 } else if (link_option(option, "as-needed", &p)) {
1373 ignoring = 1;
1374 } else if (link_option(option, "O", &p)) {
1375 ignoring = 1;
1376 } else if (link_option(option, "export-all-symbols", &p)) {
1377 s->rdynamic = 1;
1378 } else if (link_option(option, "rpath=", &p)) {
1379 copy_linker_arg(&s->rpath, p, ':');
1380 } else if (link_option(option, "enable-new-dtags", &p)) {
1381 s->enable_new_dtags = 1;
1382 } else if (link_option(option, "section-alignment=", &p)) {
1383 s->section_align = strtoul(p, &end, 16);
1384 } else if (link_option(option, "soname=", &p)) {
1385 copy_linker_arg(&s->soname, p, 0);
1386#ifdef TCC_TARGET_PE
1387 } else if (link_option(option, "large-address-aware", &p)) {
1388 s->pe_characteristics |= 0x20;
1389 } else if (link_option(option, "file-alignment=", &p)) {
1390 s->pe_file_align = strtoul(p, &end, 16);
1391 } else if (link_option(option, "stack=", &p)) {
1392 s->pe_stack_size = strtoul(p, &end, 10);
1393 } else if (link_option(option, "subsystem=", &p)) {
1394#if defined(TCC_TARGET_I386) || defined(TCC_TARGET_X86_64)
1395 if (!strcmp(p, "native")) {
1396 s->pe_subsystem = 1;
1397 } else if (!strcmp(p, "console")) {
1398 s->pe_subsystem = 3;
1399 } else if (!strcmp(p, "gui") || !strcmp(p, "windows")) {
1400 s->pe_subsystem = 2;
1401 } else if (!strcmp(p, "posix")) {
1402 s->pe_subsystem = 7;
1403 } else if (!strcmp(p, "efiapp")) {
1404 s->pe_subsystem = 10;
1405 } else if (!strcmp(p, "efiboot")) {
1406 s->pe_subsystem = 11;
1407 } else if (!strcmp(p, "efiruntime")) {
1408 s->pe_subsystem = 12;
1409 } else if (!strcmp(p, "efirom")) {
1410 s->pe_subsystem = 13;
1411#elif defined(TCC_TARGET_ARM)
1412 if (!strcmp(p, "wince")) {
1413 s->pe_subsystem = 9;
1414#endif
1415 } else
1416 goto err;
1417#endif
1418 } else if (ret = link_option(option, "?whole-archive", &p), ret) {
1419 s->alacarte_link = ret < 0;
1420 } else if (p) {
1421 return 0;
1422 } else {
1423 err:
1424 tcc_error("unsupported linker option '%s'", option);
1425 }
1426
1427 if (ignoring && s->warn_unsupported)
1428 tcc_warning("unsupported linker option '%s'", option);
1429
1430 option = skip_linker_arg(&p);
1431 }
1432 return 1;
1433}

1610static int args_parser_make_argv(const char *r, int *argc, char ***argv)
1611{
1612 int ret = 0, q, c;
1613 CString str;
1614 for(;;) {
1615 while (c = (unsigned char)*r, c && c <= ' ')
1616 ++r;
1617 if (c == 0)
1618 break;
1619 q = 0;
1620 cstr_new(&str);
1621 while (c = (unsigned char)*r, c) {
1622 ++r;
1623 if (c == '\\' && (*r == '"' || *r == '\\')) {
1624 c = *r++;
1625 } else if (c == '"') {
1626 q = !q;
1627 continue;
1628 } else if (q == 0 && c <= ' ') {
1629 break;
1630 }
1631 cstr_ccat(&str, c);
1632 }
1633 cstr_ccat(&str, 0);
1634 //printf("<%s>\n", str.data), fflush(stdout);
1635 dynarray_add(argv, argc, tcc_strdup(str.data));
1636 cstr_free(&str);
1637 ++ret;
1638 }
1639 return ret;
1640}

1642/* read list file */
1643static void args_parser_listfile(TCCState *s,
1644 const char *filename, int optind, int *pargc, char ***pargv)
1645{
1646 int fd, i;
1647 size_t len;
1648 char *p;
1649 int argc = 0;
1650 char **argv = NULL;
1651
1652 fd = open(filename, O_RDONLY | O_BINARY);
1653 if (fd < 0)
1654 tcc_error("listfile '%s' not found", filename);
1655
1656 len = lseek(fd, 0, SEEK_END);
1657 p = tcc_malloc(len + 1), p[len] = 0;
1658 lseek(fd, 0, SEEK_SET), read(fd, p, len), close(fd);
1659
1660 for (i = 0; i < *pargc; ++i)
1661 if (i == optind)
1662 args_parser_make_argv(p, &argc, &argv);
1663 else
1664 dynarray_add(&argv, &argc, tcc_strdup((*pargv)[i]));
1665
1666 tcc_free(p);
1667 dynarray_reset(&s->argv, &s->argc);
1668 *pargc = s->argc = argc, *pargv = s->argv = argv;
1669}

1671PUB_FUNC int tcc_parse_args(TCCState *s, int *pargc, char ***pargv, int optind)
1672{
1673 const TCCOption *popt;
1674 const char *optarg, *r;
1675 const char *run = NULL;
1676 int last_o = -1;
1677 int x;
1678 CString linker_arg; /* collect -Wl options */
1679 int tool = 0, arg_start = 0, noaction = optind;
1680 char **argv = *pargv;
1681 int argc = *pargc;
1682
1683 cstr_new(&linker_arg);
1684
1685 while (optind < argc) {
1686 r = argv[optind];
1687 if (r[0] == '@' && r[1] != '\0') {
1688 args_parser_listfile(s, r + 1, optind, &argc, &argv);
1689 continue;
1690 }
1691 optind++;
1692 if (tool) {
1693 if (r[0] == '-' && r[1] == 'v' && r[2] == 0)
1694 ++s->verbose;
1695 continue;
1696 }
1697reparse:
1698 if (r[0] != '-' || r[1] == '\0') {
1699 if (r[0] != '@') /* allow "tcc file(s) -run @ args ..." */
1700 args_parser_add_file(s, r, s->filetype);
1701 if (run) {
1702 tcc_set_options(s, run);
1703 arg_start = optind - 1;
1704 break;
1705 }
1706 continue;
1707 }
1708
1709 /* find option in table */
1710 for (popt = tcc_options; ; ++popt) {
1711 const char *p1 = popt->name;
1712 const char *r1 = r + 1;
1713 if (p1 == NULL)
1714 tcc_error("invalid option -- '%s'", r);
1715 if (!strstart(p1, &r1))
1716 continue;
1717 optarg = r1;
1718 if (popt->flags & TCC_OPTION_HAS_ARG) { /* r1option */
1719 if (*r1 == '\0' && !(popt->flags & TCC_OPTION_NOSEP)) {
1720 if (optind >= argc)
1721 arg_err:
1722 tcc_error("argument to '%s' is missing", r);
1723 optarg = argv[optind++];
1724 }
1725 } else if (*r1 != '\0') /* r1option */
1726 continue;
1727 break;
1728 }
1729
1730 switch (popt->index) {
1731 case TCC_OPTION_HELP:
1732 return OPT_HELP;
1733 case TCC_OPTION_HELP2:
1734 return OPT_HELP2;
1735 case TCC_OPTION_I:
1736 tcc_add_include_path(s, optarg);
1737 break;
1738 case TCC_OPTION_D:
1739 parse_option_D(s, optarg);
1740 break;
1741 case TCC_OPTION_U:
1742 tcc_undefine_symbol(s, optarg);
1743 break;
1744 case TCC_OPTION_L:
1745 tcc_add_library_path(s, optarg);
1746 break;
1747 case TCC_OPTION_B:
1748 /* set tcc utilities path (mainly for tcc development) */
1749 tcc_set_lib_path(s, optarg);
1750 break;
1751 case TCC_OPTION_l:
1752 args_parser_add_file(s, optarg, AFF_TYPE_LIB);
1753 s->nb_libraries++;
1754 break;
1755 case TCC_OPTION_pthread:
1756 parse_option_D(s, "_REENTRANT");
1757 s->option_pthread = 1;
1758 break;
1759 case TCC_OPTION_bench:
1760 s->do_bench = 1;
1761 break;
1762#ifdef CONFIG_TCC_BACKTRACE
1763 case TCC_OPTION_bt:
1764 tcc_set_num_callers(atoi(optarg));
1765 break;
1766#endif
1767#ifdef CONFIG_TCC_BCHECK
1768 case TCC_OPTION_b:
1769 s->do_bounds_check = 1;
1770 s->do_debug = 1;
1771 break;
1772#endif
1773 case TCC_OPTION_g:
1774 s->do_debug = 1;
1775 break;
1776 case TCC_OPTION_c:
1777 x = TCC_OUTPUT_OBJ;
1778 set_output_type:
1779 if (s->output_type)
1780 tcc_warning("-%s: overriding compiler action already specified", popt->name);
1781 s->output_type = x;
1782 break;
1783 case TCC_OPTION_d:
1784 if (*optarg == 'D')
1785 s->dflag = 3;
1786 else if (*optarg == 'M')
1787 s->dflag = 7;
1788 else if (*optarg == 't')
1789 s->dflag = 16;
1790 else if (isnum(*optarg))
1791 g_debug = atoi(optarg);
1792 else
1793 goto unsupported_option;
1794 break;
1795 case TCC_OPTION_static:
1796 s->static_link = 1;
1797 break;
1798 case TCC_OPTION_std:
1799 /* silently ignore, a current purpose:
1800 allow to use a tcc as a reference compiler for "make test" */
1801 break;
1802 case TCC_OPTION_shared:
1803 x = TCC_OUTPUT_DLL;
1804 goto set_output_type;
1805 case TCC_OPTION_soname:
1806 s->soname = tcc_strdup(optarg);
1807 break;
1808 case TCC_OPTION_o:
1809 if (s->outfile) {
1810 tcc_warning("multiple -o option");
1811 tcc_free(s->outfile);
1812 }
1813 s->outfile = tcc_strdup(optarg);
1814 break;
1815 case TCC_OPTION_r:
1816 /* generate a .o merging several output files */
1817 s->option_r = 1;
1818 x = TCC_OUTPUT_OBJ;
1819 goto set_output_type;
1820 case TCC_OPTION_isystem:
1821 tcc_add_sysinclude_path(s, optarg);
1822 break;
1823 case TCC_OPTION_include:
1824 dynarray_add(&s->cmd_include_files,
1825 &s->nb_cmd_include_files, tcc_strdup(optarg));
1826 break;
1827 case TCC_OPTION_nostdinc:
1828 s->nostdinc = 1;
1829 break;
1830 case TCC_OPTION_nostdlib:
1831 s->nostdlib = 1;
1832 break;
1833 case TCC_OPTION_run:
1834#ifndef TCC_IS_NATIVE
1835 tcc_error("-run is not available in a cross compiler");
1836#endif
1837 run = optarg;
1838 x = TCC_OUTPUT_MEMORY;
1839 goto set_output_type;
1840 case TCC_OPTION_v:
1841 do ++s->verbose; while (*optarg++ == 'v');
1842 ++noaction;
1843 break;
1844 case TCC_OPTION_f:
1845 if (set_flag(s, options_f, optarg) < 0)
1846 goto unsupported_option;
1847 break;
1848#ifdef TCC_TARGET_ARM
1849 case TCC_OPTION_mfloat_abi:
1850 /* tcc doesn't support soft float yet */
1851 if (!strcmp(optarg, "softfp")) {
1852 s->float_abi = ARM_SOFTFP_FLOAT;
1853 tcc_undefine_symbol(s, "__ARM_PCS_VFP");
1854 } else if (!strcmp(optarg, "hard"))
1855 s->float_abi = ARM_HARD_FLOAT;
1856 else
1857 tcc_error("unsupported float abi '%s'", optarg);
1858 break;
1859#endif
1860 case TCC_OPTION_m:
1861 if (set_flag(s, options_m, optarg) < 0) {
1862 if (x = atoi(optarg), x != 32 && x != 64)
1863 goto unsupported_option;
1864 if (PTR_SIZE != x/8)
1865 return x;
1866 ++noaction;
1867 }
1868 break;
1869 case TCC_OPTION_W:
1870 if (set_flag(s, options_W, optarg) < 0)
1871 goto unsupported_option;
1872 break;
1873 case TCC_OPTION_w:
1874 s->warn_none = 1;
1875 break;
1876 case TCC_OPTION_rdynamic:
1877 s->rdynamic = 1;
1878 break;
1879 case TCC_OPTION_Wl:
1880 if (linker_arg.size)
1881 --linker_arg.size, cstr_ccat(&linker_arg, ',');
1882 cstr_cat(&linker_arg, optarg, 0);
1883 if (tcc_set_linker(s, linker_arg.data))
1884 cstr_free(&linker_arg);
1885 break;
1886 case TCC_OPTION_Wp:
1887 r = optarg;
1888 goto reparse;
1889 case TCC_OPTION_E:
1890 x = TCC_OUTPUT_PREPROCESS;
1891 goto set_output_type;
1892 case TCC_OPTION_P:
1893 s->Pflag = atoi(optarg) + 1;
1894 break;
1895 case TCC_OPTION_MD:
1896 s->gen_deps = 1;
1897 break;
1898 case TCC_OPTION_MF:
1899 s->deps_outfile = tcc_strdup(optarg);
1900 break;
1901 case TCC_OPTION_dumpversion:
1902 printf ("%s\n", TCC_VERSION);
1903 exit(0);
1904 break;
1905 case TCC_OPTION_x:
1906 if (*optarg == 'c')
1907 s->filetype = AFF_TYPE_C;
1908 else if (*optarg == 'a')
1909 s->filetype = AFF_TYPE_ASMPP;
1910 else if (*optarg == 'n')
1911 s->filetype = AFF_TYPE_NONE;
1912 else
1913 tcc_warning("unsupported language '%s'", optarg);
1914 break;
1915 case TCC_OPTION_O:
1916 last_o = atoi(optarg);
1917 break;
1918 case TCC_OPTION_print_search_dirs:
1919 x = OPT_PRINT_DIRS;
1920 goto extra_action;
1921 case TCC_OPTION_impdef:
1922 x = OPT_IMPDEF;
1923 goto extra_action;
1924 case TCC_OPTION_ar:
1925 x = OPT_AR;
1926 extra_action:
1927 arg_start = optind - 1;
1928 if (arg_start != noaction)
1929 tcc_error("cannot parse %s here", r);
1930 tool = x;
1931 break;
1932 case TCC_OPTION_traditional:
1933 case TCC_OPTION_pedantic:
1934 case TCC_OPTION_pipe:
1935 case TCC_OPTION_s:
1936 /* ignored */
1937 break;
1938 default:
1939unsupported_option:
1940 if (s->warn_unsupported)
1941 tcc_warning("unsupported option '%s'", r);
1942 break;
1943 }
1944 }
1945 if (last_o > 0)
1946 tcc_define_symbol(s, "__OPTIMIZE__", NULL);
1947 if (linker_arg.size) {
1948 r = linker_arg.data;
1949 goto arg_err;
1950 }
1951 *pargc = argc - arg_start;
1952 *pargv = argv + arg_start;
1953 if (tool)
1954 return tool;
1955 if (optind != noaction)
1956 return 0;
1957 if (s->verbose == 2)
1958 return OPT_PRINT_DIRS;
1959 if (s->verbose)
1960 return OPT_V;
1961 return OPT_HELP;
1962}
